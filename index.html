tube_stations <- left_join(tube_stations, lines, by = c("name" = "Station"))

tube_stations <- tube_stations %>%
  filter(Operator == "London Underground") %>%
  filter(Line == "Central") %>%
  mutate(colour = case_when(Line == "Central" ~ "red"))

hainault_loop <- c(
  "Roding Valley","Chigwell","Grange Hill","Hainault",
  "Fairlop","Barkingside","Newbury Park","Gants Hill","Redbridge","Wanstead"
)


# Main branch line
tube_line_main <- tube_stations %>%
  filter(name %in% main_branch) %>%
  mutate(order = match(name, main_branch)) %>%
  arrange(order) %>%
  summarise(
    geometry = st_sfc(st_linestring(st_coordinates(.))),
    .groups = "drop"
  )

# Hainault loop line
tube_line_loop <- tube_stations %>%
  filter(name %in% hainault_loop) %>%
  mutate(order = match(name, hainault_loop)) %>%
  arrange(order) %>%
  summarise(
    geometry = st_sfc(st_linestring(st_coordinates(.))),
    .groups = "drop"
  )

# Combine
tube_lines <- bind_rows(
  tube_line_main %>% mutate(branch = "Main"),
  tube_line_loop %>% mutate(branch = "Hainault loop")
)

# convert ofsted coords:
ofsted <- ofsted %>%
  na.omit() %>%
  st_as_sf(coords = c("easting", "northing"), crs = 27700) %>%
  st_transform(4326)



# Only plot schools close to central line
x_miles <- 2   # example
buffer <- st_buffer(tube_stations, dist = x_miles * 1609.34)

# make ofsted_sf
ofsted_sf <- st_as_sf(
  ofsted, 
  coords = c("easting", "northing"),  # the two columns in your df
  crs = 27700                          # EPSG code for BNG
)

# Filter Ofsted schools within that buffer
ofsted_near <- ofsted_sf %>%
  st_filter(buffer, .predicate = st_within)
#Transform back to WGS84 for leaflet
ofsted_near <- st_transform(ofsted_near, 4326)

# Define grade levels and palette
grade_levels <- c("other", "requires improvement", "inadequate", 
                  "satisfactory", "good", "outstanding")
grade_palette <- c("grey", "brown", "brown", "orange", "green", "blue")

# Make sure 'grade' is a factor with the correct order
ofsted_sf$grade <- factor(ofsted_sf$grade, levels = grade_levels)

# Create the color palette with explicit ordering
pal <- colorFactor(palette = grade_palette, domain = grade_levels, ordered = TRUE)

# Use the same ordered levels for legend
legend_values <- grade_levels

  ) %>%
